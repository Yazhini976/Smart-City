CREATE DATABASE smart_city;

USERS TABLE

CREATE TABLE users (

    phone_number VARCHAR(15) PRIMARY KEY,

    name VARCHAR(100),

     user_id serial

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


modules table

CREATE TABLE modules (

    module_id SERIAL PRIMARY KEY,

    module_name VARCHAR(100) UNIQUE NOT NULL
);

field officer table

CREATE TABLE field_officers (

    officer_id SERIAL PRIMARY KEY,

    module_id INT NOT NULL
    REFERENCES modules(module_id),

    officer_name VARCHAR(100) NOT NULL,

    phone_number VARCHAR(15) UNIQUE NOT NULL,

    ward_from INT NOT NULL,

    ward_to INT NOT NULL,

    is_active BOOLEAN DEFAULT TRUE,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


ward mapping table

CREATE TABLE ward_mapping (

    mapping_id SERIAL PRIMARY KEY,

    module_id INT NOT NULL
    REFERENCES modules(module_id),

    ward_no INT NOT NULL,

    officer_id INT NOT NULL
    REFERENCES field_officers(officer_id),

    UNIQUE(module_id, ward_no)
);

complaints table

CREATE TABLE complaints (

    complaint_id UUID PRIMARY KEY
    DEFAULT gen_random_uuid(),

    user_phone VARCHAR(15)
    REFERENCES users(phone_number),

    module_id INT NOT NULL
    REFERENCES modules(module_id),

    ward_no INT NOT NULL,

    assigned_officer_id INT
    REFERENCES field_officers(officer_id),

    location TEXT NOT NULL,

    latitude DECIMAL(10,8),

    longitude DECIMAL(11,8),

    complaint_photo TEXT,

    reason VARCHAR(150),

    severity VARCHAR(20),

    ai_detected_issue VARCHAR(150),

    ai_confidence DECIMAL(5,2),

    status VARCHAR(30)
    DEFAULT 'PENDING',

    created_at TIMESTAMP
    DEFAULT CURRENT_TIMESTAMP,

    updated_at TIMESTAMP
    DEFAULT CURRENT_TIMESTAMP
);

complaint update table

CREATE TABLE complaint_updates (

    update_id SERIAL PRIMARY KEY,

    complaint_id UUID
    REFERENCES complaints(complaint_id),

    officer_id INT
    REFERENCES field_officers(officer_id),

    old_status VARCHAR(30),

    new_status VARCHAR(30),

    remarks TEXT,

    updated_at TIMESTAMP
    DEFAULT CURRENT_TIMESTAMP
);
